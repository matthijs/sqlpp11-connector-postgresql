function(
    test_emplace
    testname
    )

    set(
        target
        Test_${testname}
        )

    # Add headers to sources to enable file browsing in IDEs
    add_executable(
        ${target}
        ${testname}.cpp
        )

    target_link_libraries(
        ${target}
        sqlpp-postgresql
        ${PostgreSQL_LIBRARIES}
        )


    target_compile_features(
        ${target}
        PRIVATE
            cxx_auto_type
        )

    target_include_directories(
        ${target}
        PRIVATE
            ${PostgreSQL_INCLUDE_DIRS}
            "../src"
        )

    add_test(
        NAME
            ${testname}
        COMMAND
            ${target}
        )

endfunction()
